## Adding another pieces of state

```jsx
  const [isOpen, setIsOpen] = useState(true);


  function handleOpen() {
    setIsOpen(!isOpen);
  }

  return (
    //REACT FRAGMENT
    <>
      <button
        onClick={handleOpen}
        className="close" //
      >
        &times;
      </button>
  )
```
